DROP DATABASE IF EXISTS jediorder;
CREATE DATABASE jediorder;
USE jediorder;

CREATE TABLE Jedi (
	ID BIGINT NOT NULL AUTO_INCREMENT,
	LastName VARCHAR(30) NOT NULL,
	FirstName VARCHAR(250) NOT NULL,
	Birth DATE;
	Death DATE;
	JediRank VARCHAR(15) NOT NULL;
	SaberType VARCHAR(10) NOT NULL;
	SaberColor VARCHAR(30) NOT NULL;
	PRIMARY KEY (ID)
);


CREATE TABLE Sith (
	ID BIGINT NOT NULL AUTO_INCREMENT,
	LastName VARCHAR(30) NOT NULL,
	FirstName VARCHAR(250) NOT NULL,
	Birth DATE;
	Death DATE;
	TitleAtDeath VARCHAR(15) NOT NULL;
	SaberColor VARCHAR() NOT NULL;
	PRIMARY KEY (ID),
);


CREATE TABLE BountyHunters (
	ID BIGINT NOT NULL AUTO_INCREMENT,
	LastName VARCHAR(30) NOT NULL,
	FirstName VARCHAR(250) NOT NULL,
	Birth DATE;
	Death DATE;
	Organisation VARCHAR(250);
	PRIMARY KEY (ID)
);

CREATE TABLE Smugglers (
	ID BIGINT NOT NULL AUTO_INCREMENT,
	LastName VARCHAR(30) NOT NULL,
	FirstName VARCHAR(250) NOT NULL,
	Birth DATE;
	Death DATE;
	Organisation VARCHAR(250);
	PRIMARY KEY (ID)
);

CREATE TABLE Battles (
	BattleID BIGINT NOT NULL,
	Location VARCHAR(250) NOT NULL,
	JediA BIGINT NOT NULL,
	JediB BIGINT,
	SithA BIGINT NOT NULL,
	SithB BIGINT,
	SmugglerA BIGINT NOT NULL,
	SmugglerB BIGINT,
	BountyHunterA BIGINT NOT NULL,
	BountyHunterB BIGINT,
	Outcome VARCHAR(250) NOT NULL,
	PRIMARY KEY (BattleID),
	FOREIGN KEY(JediA) REFERENCES Jedi (ID),
	FOREIGN KEY(JediB) REFERENCES Jedi (ID),
	FOREIGN KEY(SithA) REFERENCES Sith (ID),
	FOREIGN KEY(SithB) REFERENCES Sith (ID),
	FOREIGN KEY(SmugglerA) REFERENCES Smugglers (ID),
	FOREIGN KEY(SmugglerB) REFERENCES Smugglers (ID),
	FOREIGN KEY(BountyHunterA) REFERENCES BountyHunters (ID),
	FOREIGN KEY(BountyHunterB) REFERENCES BountyHunters (ID),
);
